setMethod('Ops',
          signature(e1='trick', e2='trick'),
          function(e1, e2) {
            e1@time <- callGeneric(slot(e1, 'time'), slot(e2, 'time'))
            validObject(e1)
            return(e1)
          })

t1 <- new('trick', id = letters[1:3], time = matrix(1:9, ncol = 3))
t2 <- new('trick', id = letters[1:3], time = matrix(9:1, ncol = 3))
t1 + t2
