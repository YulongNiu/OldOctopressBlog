# 进入home路径
# cd /home/


# 进入默认postgres用户，需要输入root密码。
# 之后系统命令提示符会变成类似“bash-4.3$”。
# su postgres

# 创建用户
# -P：创建用户后立即创建密码
# -s：用户为superuser
# -e：打印消息
bash-4.3$ createuser -P -s -e Kitty

# 创建数据库
# -O：制定用户
bash-4.3$ createdb -O Kitty testdb

# 退出
bash-4.3$ exit

# 再创建新的数据库时，不需要进入postgres。
# 直接在当前登录用户下创建即可。
# 由于PostgreSQL用户与系统登录用户相同，不需要指定PostgreSQL用户
$ creatdb testdb2

# 删除数据库
$ dropdb testdb2
