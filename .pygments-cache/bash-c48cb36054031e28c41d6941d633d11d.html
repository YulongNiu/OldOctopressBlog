# 安装最快源插件
# dnf install dnf-plugin-fastestmirror

# 安装并行下载工具
# dnf install dnf-axelget

# 终止dnf安装
# rm -f /var/run/yum.pid

# 查询包
# dnf search pkg

# 重新安装包
# dnf reinstall pkg

# 升级包
# dnf update pkg

# 卸载包
# dnf remove pkg

# 查询已安装包信息
# dnf info pkg

# 查看仓库包列表
# dnf list pkg*

# 查看已安装的包
# dnf list installed
