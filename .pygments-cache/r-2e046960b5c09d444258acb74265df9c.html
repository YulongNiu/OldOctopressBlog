# NULL is neglected in vector
> c(1, NA, NULL, FALSE)
[1]  1 NA  0

# also in matrix
> matrix(c(1, NA, NULL, 2), c(2, 2))
     [,1] [,2]
[1,]    1    2
[2,]   NA    1
警告信息：
In matrix(c(1, NA, NULL, 2), c(2, 2)) : 数据长度[3]不是矩阵行数[2]的整倍

# also in list
> tmp <- list(1, 2)
> tmp[1] <- NULL
> tmp
[[1]]
[1] 2
> tmp[[1]] <- NULL
> tmp
list()
